---
title: Popover
description: A headless popover powered by @data-slot/popover.
---

```astro live
---
import {
  Popover,
  PopoverClose,
  PopoverContent,
  PopoverTrigger,
} from "@/components/ui/popover"
---

<Popover>
  <PopoverTrigger class="rounded-md border px-3 py-2 text-sm">
    Open Popover
  </PopoverTrigger>
  <PopoverContent>
    <p>Popover content here</p>
    <PopoverClose class="mt-3 rounded-md border px-3 py-2 text-sm">
      Close
    </PopoverClose>
  </PopoverContent>
</Popover>
```

## Installation

```bash
npm install @data-slot/popover
```

## Usage

```ts
import {
  Popover,
  PopoverClose,
  PopoverContent,
  PopoverPortal,
  PopoverPositioner,
  PopoverTrigger,
} from "@/components/ui/popover"
```

```astro
<Popover>
  <PopoverTrigger>Trigger</PopoverTrigger>
  <PopoverContent>
    Content
    <PopoverClose>Close</PopoverClose>
  </PopoverContent>
</Popover>
```

```astro
<Popover>
  <PopoverTrigger>Trigger</PopoverTrigger>
  <PopoverPortal>
    <PopoverPositioner>
      <PopoverContent>Content</PopoverContent>
    </PopoverPositioner>
  </PopoverPortal>
</Popover>
```
