```astro live
  SheetContent,

  SheetTrigger,

  <SheetTrigger asChild>
    <Button variant="outline">Open Sheet</Button>

  <SheetContent>
    <SheetHeader>

    </SheetHeader>

```

## Installation

To install the sheet component, run the following command:

```bash
pnpm dlx shadcn@latest add @fulldev-ui/sheet
```

## Usage

```ts
import {
  Sheet,
  SheetClose,
  SheetContent,
  SheetDescription,
  SheetFooter,
  SheetHeader,
  SheetTitle,
  SheetTrigger,
} from "@/components/ui/sheet"
```

```astro
<Sheet>
  <SheetTrigger asChild>
    <Button>Open</Button>
  </SheetTrigger>
  <SheetContent>
    <SheetHeader>
      <SheetTitle>Sheet Title</SheetTitle>
      <SheetDescription>Sheet Description</SheetDescription>
    </SheetHeader>
    <SheetFooter>
      <SheetClose asChild>
        <Button variant="outline">Close</Button>
      </SheetClose>
      <Button>Save</Button>
    </SheetFooter>
  </SheetContent>
</Sheet>
```
