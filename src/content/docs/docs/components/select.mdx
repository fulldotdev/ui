## Installation

### Command

```bash
npx shadcn@latest add @fulldev/select
```

### Manual

1. Install the following dependency:

```bash
npm install @data-slot/select
```

2. Copy and paste the select component files into `src/components/ui/select/`.

3. Update the import paths to match your project setup.

```astro

      </SelectItem>
      <SelectItem value="system">
        <SelectItemText>System</SelectItemText>
```

Wrap visible option labels with `SelectItemText` for the closest item-aligned
behavior and trigger text parity.

## Composition

Use the following composition to build a `Select`:

```text
Select
├── SelectTrigger
│   └── SelectValue
└── SelectContent
    └── SelectViewport
        ├── SelectGroup
        │   ├── SelectLabel
        │   ├── SelectItem
        │   │   └── SelectItemText
        │   └── SelectItem
        │       └── SelectItemText
        ├── SelectSeparator
        └── SelectGroup
            ├── SelectLabel
            ├── SelectItem
            │   └── SelectItemText
            └── SelectItem
                └── SelectItemText
```

## Examples

### Align Item With Trigger

Use the `position` prop on `SelectContent` to control alignment. When
`position="item-aligned"` (default), the popup positions so the selected item
appears over the trigger. When `position="popper"`, the popup aligns to the
trigger edge.

```astro live
---
import ChevronDownIcon from "lucide-static/icons/chevron-down.svg"

import {
  Select,
  SelectContent,
  SelectItem,
  SelectItemText,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select"
---

<div class="grid w-full max-w-2xl gap-6 @md:grid-cols-2">
  <div class="space-y-3">
    <p class="text-sm font-medium">Item aligned</p>
    <Select defaultValue="pro" class="max-w-xs">
      <SelectTrigger>
        <SelectValue />
        <ChevronDownIcon class="size-4 opacity-50" />
      </SelectTrigger>
      <SelectContent position="item-aligned">
        <SelectItem value="starter">
          <SelectItemText>Starter</SelectItemText>
        </SelectItem>
        <SelectItem value="pro">
          <SelectItemText>Pro</SelectItemText>
        </SelectItem>
        <SelectItem value="enterprise">
          <SelectItemText>Enterprise</SelectItemText>
        </SelectItem>
      </SelectContent>
    </Select>
  </div>

  <div class="space-y-3">
    <p class="text-sm font-medium">Popper</p>
    <Select defaultValue="pro" class="max-w-xs">
      <SelectTrigger>
        <SelectValue />
        <ChevronDownIcon class="size-4 opacity-50" />
      </SelectTrigger>
      <SelectContent position="popper" side="bottom" align="start">
        <SelectItem value="starter">
          <SelectItemText>Starter</SelectItemText>
        </SelectItem>
        <SelectItem value="pro">
          <SelectItemText>Pro</SelectItemText>
        </SelectItem>
        <SelectItem value="enterprise">
          <SelectItemText>Enterprise</SelectItemText>
        </SelectItem>
      </SelectContent>
    </Select>
  </div>
</div>
```

```astro live
          <SelectItemText>Spinach</SelectItemText>
        </SelectItem>
      </SelectGroup>
    </SelectViewport>
  </SelectContent>
</Select>
```

### Scrollable

A select with many items that scrolls.

```astro live
---
import ChevronDownIcon from "lucide-static/icons/chevron-down.svg"

import {
  Select,
  SelectContent,
  SelectItem,
  SelectItemText,
  SelectTrigger,
  SelectValue,
  SelectViewport,
} from "@/components/ui/select"

const timezones = [
  "Pacific/Honolulu",
  "America/Anchorage",
  "America/Los_Angeles",
  "America/Denver",
  "America/Chicago",
  "America/New_York",
  "America/Sao_Paulo",
  "Europe/London",
  "Europe/Amsterdam",
  "Europe/Berlin",
  "Europe/Athens",
  "Asia/Dubai",
  "Asia/Kolkata",
  "Asia/Bangkok",
  "Asia/Tokyo",
  "Australia/Sydney",
]
---

<Select class="max-w-xs" placeholder="Select timezone">
  <SelectTrigger>
    <SelectValue />
    <ChevronDownIcon class="size-4 opacity-50" />
  </SelectTrigger>
  <SelectContent class="max-h-56">
    <SelectViewport>
      {
        timezones.map((timezone) => (
          <SelectItem value={timezone}>
            <SelectItemText>{timezone}</SelectItemText>
          </SelectItem>
        ))
      }
    </SelectViewport>
  </SelectContent>
</Select>
```

### Invalid

Add the `data-invalid="true"` attribute to the `Field` component and the
`aria-invalid="true"` attribute to the `SelectTrigger` component to show an
error state.

```astro
<Field data-invalid="true">
  <FieldLabel for="fruit-invalid">Fruit</FieldLabel>
  <Select placeholder="Select a fruit">
    <SelectTrigger id="fruit-invalid" aria-invalid="true">
      <SelectValue />
    </SelectTrigger>
    <SelectContent>
      <SelectItem value="apple">
        <SelectItemText>Apple</SelectItemText>
      </SelectItem>
    </SelectContent>
  </Select>
</Field>
```

mode.

```astro live

  SelectContent,
```
