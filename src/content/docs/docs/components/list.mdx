---
title: List
description: A container for displaying a series of items in a list format.
---

```astro live
---
import { List } from "@/components/ui/list"
---

<List class="w-full max-w-sm">
  <li>First item</li>
  <li>Second item</li>
  <li>Third item</li>
  <li>Fourth item</li>
</List>
```

## Installation

To install the list component, run the following command:

```bash
pnpm dlx shadcn@latest add @fulldev-ui/list
```

## Usage

```ts
import { List } from "@/components/ui/list"
```

```astro
<List>
  <li>Item</li>
</List>
```
