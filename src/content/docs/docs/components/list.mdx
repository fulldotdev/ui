---
title: List
description: A container for displaying a series of items in a list format.
---

```astro live
---
import { List, ListItem } from "@/components/ui/list"
---

<List class="w-full max-w-sm">
  <ListItem>First item</ListItem>
  <ListItem>Second item</ListItem>
  <ListItem>Third item</ListItem>
  <ListItem>Fourth item</ListItem>
</List>
```

## Installation

To install the list component, run the following command:

```bash
pnpm dlx shadcn@latest add @fulldev-ui/list
```

## Usage

```ts
import { List, ListItem } from "@/components/ui/list"
```

```astro
<List>
  <ListItem>Item 1</ListItem>
  <ListItem>Item 2</ListItem>
  <ListItem>Item 3</ListItem>
</List>
```
