## Installation

To install the native carousel component, run the following command:

```bash
npx shadcn@latest add @fulldev-ui/native-carousel
```

## Usage

```ts
import {
  NativeCarousel,
  NativeCarouselContent,
  NativeCarouselItem,
  NativeCarouselNext,
  NativeCarouselPrevious,
} from "@/components/ui/native-carousel"
```

```astro
<NativeCarousel>
  <NativeCarouselContent>
    <NativeCarouselItem>
      <div>Slide 1</div>
    </NativeCarouselItem>
    <NativeCarouselItem>
      <div>Slide 2</div>
    </NativeCarouselItem>
  </NativeCarouselContent>
  <NativeCarouselPrevious />
  <NativeCarouselNext />
</NativeCarousel>
```
