### header-1

```astro live
---
import placeholderImage from "@/assets/placeholder.png"
import HeaderBlock from "@/components/blocks/header-1.astro"

const global = {
  name: "Fulldev",
  logo: { dark: placeholderImage, light: placeholderImage },
  header: {
    links: [
      { text: "Docs", href: "/docs/" },
      { text: "Components", href: "/docs/components/" },
      { text: "Blocks", href: "/blocks/" },
    ],
    buttons: [{ text: "Start Project", href: "/docs/" }],
  },
}
---

<HeaderBlock
  props={{
    global,
    headerLinks: global.header.links,
    headerButtons: global.header.buttons.map((button, index) => ({
      href: button.href,
      label: button.text,
      variant: index === 0 ? "default" : "ghost",
    })),
    homeHref: "/",
    github: {
      href: "https://github.com/fulldotdev/ui",
      ariaLabel: "GitHub repository",
      count: "557",
    },
    mobileMenuLabel: "Toggle navigation",
  }}
 />
```
